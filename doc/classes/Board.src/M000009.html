        def placeMines()
                placed = 0
                while placed < @mines do
                        column = rand(@columns)
                        row = rand(@rows)
                        if @board[row][column].value != 'M'
                                @board[row][column].value = 'M'
                                @board[row][column].increaseNeighbors()
                                placed += 1
                        end
                end
        end
